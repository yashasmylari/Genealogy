


CREATE TABLE person (
	id INT AUTO_INCREMENT PRIMARY KEY,
	name TEXT,
	firstName TEXT,
	lastName TEXT,
	dateOfBirth TEXT,
	placeOfBirth TEXT,
	isAlive TINYINT(1) DEFAULT 1,
	dateOfDeath TEXT,
	imageUrl TEXT,
	imageData TEXT,
	residence TEXT,
	nationality TEXT,
	religion TEXT,
	clan TEXT,
	ethnicity TEXT,
	occupation TEXT,
	education TEXT,
	deleteFlag TINYINT(1) DEFAULT 0
);



CREATE TABLE locations (
	id INT AUTO_INCREMENT PRIMARY KEY,
	name TEXT,
	deleteFlag TINYINT(1) DEFAULT 0
);


CREATE TABLE person_locations (
	id INT AUTO_INCREMENT PRIMARY KEY,
	personId INT,
	locationId INT,
	deleteFlag TINYINT(1) DEFAULT 0,
	FOREIGN KEY (personId) REFERENCES person(id),
	FOREIGN KEY (locationId) REFERENCES locations(id)
);



CREATE TABLE languages (

	FOREIGN KEY (locationId) REFERENCES languages(id)
);
